    Query Bing web search with improved headers and fallback.

    count= requests more results than needed to account for filtered items.
    mkt= sets the market/region.
    """
    # Apply backend-specific strategies
    cfg = _backend_specific_strategy(cfg)
    
    params = urllib.parse.urlencode({
        "q":     cfg.query,
        "count": min(cfg.limit * 2, 50),
        "mkt":   cfg.region,
        "safeSearch": cfg.safe.capitalize(),
        "setlang":    cfg.lang,
    })

    logger.debug("Bing search: %r region=%s", cfg.query, cfg.region)
    
    # Add adaptive delay before request (Bing is more aggressive)
    time.sleep(_adaptive_delay(cfg, 0, 0.6))
    
    # Try with enhanced headers for better compatibility
    original_headers = cfg.headers.copy()
    cfg.headers.extend([
        "Accept: text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,image/apng,*/*;q=0.8",
        "Accept-Language: en-US,en;q=0.9",
        "Referer: https://www.bing.com/",
        "Sec-Fetch-Dest: document",
        "Sec-Fetch-Mode: navigate",
        "Sec-Fetch-Site: same-origin",
    ])
    
    html = _http_get(f"https://www.bing.com/search?{params}", cfg, ua_idx=1)
    
    # Restore original headers
    cfg.headers = original_headers
    
    if not html:
        logger.warning("Bing returned empty response")
        return []

    results = _parse_bing_html(html, cfg.limit)
    logger.debug("Bing returned %d results", len(results))
    return results

def _parse_searx_html(html: str, limit: int, base_url: str) -> List[SearchResult]:
    """
    Parse Searx HTML results page.

    Searx wraps each result in <article class="result ..."> blocks.
    Relative URLs are resolved against the base_url.
    """
    results: List[SearchResult] = []
    seen:    set                = set()

    # Search blocks - articles with result class
    blocks = re.findall(
        r'<article[^>]*class=["\'][^"\']*\bresult\b[^"\']*["\'][^>]*>.*?</article>',
        html, re.DOTALL,
    )

    for block in blocks:
        if len(results) >= limit:
            break

        # Extract title and URL
        # Pattern 1: h3 > a
        # Pattern 2: a[class~=result_header]
        m = re.search(
            r'<h[23][^>]*>.*?<a[^>]*href=["\']([^"\']+)["\'][^>]*>(.*?)</a>.*?</h[23]>',
            block, re.DOTALL,
        ) or re.search(
            r'<a[^>]*class=["\'][^"\']*\bresult_header\b[^"\']*["\'][^>]*href=["\']([^"\']+)["\'][^>]*>(.*?)</a>',
            block, re.DOTALL,
        )
        
        if not m:
            continue

        raw_url = m.group(1)
        title   = _clean_html(m.group(2))

        # Resolve relative URLs
        url = urllib.parse.urljoin(base_url, raw_url)

        if not url.startswith("http") or url in seen:
            continue
        seen.add(url)

        # Snippet - try multiple patterns
        sm = re.search(
            r'<p[^>]*class=["\'](?:[^"\']*\b)?content\b(?:[^"\']*)?["\'][^>]*>(.*?)</p>',
            block, re.DOTALL,
        ) or re.search(
            r'<div[^>]*class=["\'](?:[^"\']*\b)?result_content\b(?:[^"\']*)?["\'][^>]*>(.*?)</div>',
            block, re.DOTALL,
        )
        snippet = _clean_html(sm.group(1)) if sm else ""

        results.append(SearchResult(title=title, url=url, snippet=snippet))

    return results[:limit]

def _parse_google_html(html: str, limit: int, base_url: str) -> List[SearchResult]:
    """
    Parse Google HTML results page.

    Google wraps results in divs with class 'g'.
    Relative URLs are resolved against the base_url.
    """
    results: List[SearchResult] = []
    seen:    set                = set()

    # Find result blocks. Google uses divs with class 'g' or 'tfo-unorganized-results'
    blocks = re.findall(
        r'<div[^>]*class=["\'](?:[^"\']*\b)?g\b(?:[^"\']*)["\'][^>]*>(.*?)</div>',
        html, re.DOTALL
    ) or re.findall(
        r'<div[^>]*class=["\'](?:[^"\']*\b)?tfo-unorganized-results\b(?:[^"\']*)["\'][^>]*>(.*?)</div>',
        html, re.DOTALL
    )

    for block in blocks:
        if len(results) >= limit:
            break

        # Extract title and URL
        # Title is usually in an <h3> tag, URL in an <a> tag within it.
        m = re.search(
            r'<h3[^>]*>.*?<a[^>]*href=["\']([^"\']+)["\'][^>]*>(.*?)</a>.*?</h3>',
            block, re.DOTALL
        )
        if not m:
            continue

        raw_url = m.group(1)
        title   = _clean_html(m.group(2))

        # Resolve relative URLs if necessary
        url = urllib.parse.urljoin(base_url, raw_url)

        # Skip non-HTTP URLs, ads, and duplicates
        if not url.startswith("http") or url in seen:
            continue
        seen.add(url)

        # Extract snippet - typically in a <p> tag within the result block
        sm = re.search(
            r'<p[^>]*>(.*?)</p>',
            block, re.DOTALL
        )
        snippet = _clean_html(sm.group(1)) if sm else ""

        results.append(SearchResult(title=title, url=url, snippet=snippet))

    return results[:limit]


def _search_google(cfg: SearchConfig) -> List[SearchResult]:
    """
    Query Google Search HTML endpoint.

    Uses parameters like `q`, `hl`, `gl`, `num`, `start`, `safe`.
    """
    # Apply backend-specific strategies
    cfg = _backend_specific_strategy(cfg)
    
    # Google uses different parameters for safe search
    safe_map = {"off": "off", "moderate": "medium", "strict": "strict"}
    
    params = urllib.parse.urlencode({
        "q":     cfg.query,
        "hl":    cfg.lang.split('-')[0],  # Use language code only (e.g., 'en')
        "gl":    cfg.region.split('-')[1] if '-' in cfg.region else cfg.region, # Use region code (e.g., 'us')
        "num":   cfg.limit,
        "start": 0, # Start from the first result for simplicity
        "safe":  safe_map.get(cfg.safe, "medium"),
    })

    # Try multiple Google instances for better reliability
    google_instances = _GOOGLE_SEARCH_URLS
    
    html = None
    successful_endpoint = None

    for i, base in enumerate(google_instances):
        logger.debug("Google search: %r instance=%s", cfg.query, base)
        
        # Add adaptive delay
        time.sleep(_adaptive_delay(cfg, i, 0.7))
        
        search_url = f"{base}?{params}"
        html = _http_get(search_url, cfg, ua_idx=i)
        if html:
            successful_endpoint = base
            break
        else:
            logger.debug("Google %s returned empty response", base)

    if not html:
        logger.warning("All Google instances failed")
        return []

    results = _parse_google_html(html, cfg.limit, successful_endpoint)
    logger.debug("Google returned %d results from %s", len(results), successful_endpoint)
    return results

def crawl_url(url: str, cfg: SearchConfig, ua_idx: int = 0) -> Dict[str, Any]:
    """
    Fetch a URL and extract its readable text content.

    Uses stdlib HTMLParser (_TextExtractor) — no external dependencies.

    Returns a dict with keys:
      title (str), content (str), content_length (int),
      truncated (bool), error (str)
    """
    logger.debug("Crawling: %s", url)
    html = _http_get(url, cfg, ua_idx=ua_idx)

    if html is None:
        return {"title": "", "content": "", "content_length": 0,
                "truncated": False, "error": "Failed to fetch URL"}

    title, text = _extract_text(html)

    truncated = False
    if len(text) > cfg.content_length:
        text      = text[:cfg.content_length] + "..."
        truncated = True

    return {
        "title":          title,
        "content":        text,
        "content_length": len(text),
        "truncated":      truncated,
        "error":          "",
    }


def download_url(url: str, cfg: SearchConfig, ua_idx: int = 0) -> Dict[str, Any]:
    """
    Download a URL to a local file in the current directory.

    Filename is derived from the URL path; falls back to a timestamp-based
    name when the URL has no useful path component.

    Returns a dict with keys:
      downloaded_file (str), file_size (int), error (str)
    """
    logger.debug("Downloading: %s", url)

    # Derive filename from URL path
    parsed   = urllib.parse.urlparse(url)
    basename = Path(parsed.path).name
    # Strip query string artifacts and sanitise
    basename = re.sub(r"[^\w.\-]", "_", basename.split("?")[0])
    if not basename or basename == "_":
        basename = f"download_{int(time.time())}"

    data = _http_get_binary(url, cfg, ua_idx=ua_idx)
    if data is None:
        return {"downloaded_file": "", "file_size": 0, "error": "Download failed"}

    out_path = Path(basename)
    # Avoid overwriting existing files
    stem, suffix = out_path.stem, out_path.suffix
    counter = 1
    while out_path.exists():
        out_path = Path(f"{stem}_{counter}{suffix}")
        counter += 1

    try:
        out_path.write_bytes(data)
        return {
            "downloaded_file": str(out_path),
            "file_size":       len(data),
            "error":           "",
        }
    except OSError as exc:
        return {"downloaded_file": "", "file_size": 0, "error": str(exc)}


def _randomize_request_timing(cfg: SearchConfig, request_count: int) -> float:
    """Add random delays to mimic human behavior with enhanced patterns"""
    import random
    import time
    
    # Base delay with random variation
    base_delay = cfg.delay
    
    # Human-like timing patterns
    # 1. Reading time simulation (humans don't click instantly)
    if request_count > 0:
        # Simulate reading time between clicks
        reading_time = random.uniform(1.0, 4.0)  # 1-4 seconds to "read"
        base_delay += reading_time
    
    # 2. Fatigue factor (humans slow down over time)
    if request_count > 10:
        fatigue_factor = 1.0 + (request_count - 10) * 0.1  # 10% slower per 10 requests
        base_delay *= fatigue_factor
    
    # 3. Random human-like pauses
    if random.random() < 0.1:  # 10% chance of a longer pause
        pause_time = random.uniform(5.0, 15.0)  # 5-15 second pause
        base_delay += pause_time
    
    # 4. Time-of-day based behavior
    current_hour = time.localtime().tm_hour
    if 9 <= current_hour <= 17:  # Business hours - more active
        activity_factor = random.uniform(0.7, 1.2)
    elif 22 <= current_hour or current_hour <= 6:  # Late night - less active
        activity_factor = random.uniform(1.3, 2.5)
    else:  # Evening hours
        activity_factor = random.uniform(0.9, 1.8)
    
    base_delay *= activity_factor
    
    # 5. Random variation (mouse movement, thinking, etc.)
    random_factor = random.uniform(0.7, 2.0)
    
    # 6. Weekend vs weekday behavior
    weekday = time.localtime().tm_wday  # 0=Monday, 6=Sunday
    if weekday >= 5:  # Weekend
        weekend_factor = random.uniform(1.2, 2.0)  # Slower on weekends
        base_delay *= weekend_factor
    
    # 7. Burst behavior (humans sometimes click rapidly)
    if request_count > 0 and random.random() < 0.05:  # 5% chance
        # Rapid burst of 2-3 requests
        burst_factor = random.uniform(0.2, 0.5)
        base_delay *= burst_factor
    
    # Ensure minimum delay
    min_delay = 0.5
    final_delay = max(base_delay * random_factor, min_delay)
    
    # Cap maximum delay to prevent excessive waiting
    max_delay = 30.0
    return min(final_delay, max_delay)

def _enrich_results(
    results: List[SearchResult],
    cfg:     SearchConfig,
) -> List[SearchResult]:
    """
    Crawl or download each result URL in parallel using a thread pool.

    Thread count is bounded by cfg.threads and len(results).
    Inter-request delay (cfg.delay) is applied per-thread with randomization.
    """
    if not results:
        return results

    workers = min(cfg.threads, len(results))

    def _process(idx_result: Tuple[int, SearchResult]) -> SearchResult:
        idx, result = idx_result
        if idx > 0:
            # Use randomized timing to mimic human behavior
            delay = _randomize_request_timing(cfg, idx)
            time.sleep(delay)

        if cfg.do_crawl:
            info = crawl_url(result.url, cfg, ua_idx=idx)
            result.page_title     = info["title"]
            result.content        = info["content"]
            result.content_length = info["content_length"]
            result.truncated      = info["truncated"]
            result.crawl_error    = info["error"]

        elif cfg.do_download:
            info = download_url(result.url, cfg, ua_idx=idx)
            result.downloaded_file = info["downloaded_file"]
            result.file_size       = info["file_size"]
            result.download_error  = info["error"]

        return result

    enriched: List[SearchResult] = [None] * len(results)  # type: ignore[list-item]

    with ThreadPoolExecutor(max_workers=workers) as pool:
        futures = {
            pool.submit(_process, (i, r)): i
            for i, r in enumerate(results)
        }
        for future in as_completed(futures):
            i = futures[future]
            try:
                enriched[i] = future.result()
            except Exception as exc:
                logger.debug("Enrichment failed for result %d: %s", i, exc)
                enriched[i] = results[i]

    return enriched

def format_results(results: List[SearchResult], fmt: str) -> str:
    """
    Convert a list of SearchResult objects to the requested output format.

    fmt: 'json' | 'jsonl' | 'markdown' | 'csv' | 'tsv'

    Raises ValueError for unknown formats.
    """
    dicts = [r.to_dict() for r in results]

    if fmt == "json":
        return json.dumps(dicts, indent=2, ensure_ascii=False)

    if fmt == "jsonl":
        # One compact JSON object per line
        return "\n".join(json.dumps(d, ensure_ascii=False) for d in dicts)

    if fmt == "markdown":
        return _format_markdown(results)

    if fmt == "csv":
        return _format_dsv(results, delimiter=",")

    if fmt == "tsv":
        return _format_dsv(results, delimiter="\t")

    raise ValueError(f"Unknown format '{fmt}'. Valid: {', '.join(VALID_FORMATS)}")


def _format_markdown(results: List[SearchResult]) -> str:
    """Render results as a Markdown document."""
    lines: List[str] = []
    for i, r in enumerate(results, 1):
        lines.append(f"## {i}. {r.title or 'Untitled'}")
        lines.append(f"**URL:** {r.url}")
        if r.snippet:
            lines.append(f"\n> {r.snippet}")
        if r.content:
            lines.append(f"\n### Content\n{r.content}")
        if r.downloaded_file:
            lines.append(f"\n**Downloaded:** `{r.downloaded_file}` ({r.file_size:,} bytes)")
        if r.crawl_error:
            lines.append(f"\n*Crawl error: {r.crawl_error}*")
        if r.download_error:
            lines.append(f"\n*Download error: {r.download_error}*")
        lines.append("\n---")
    return "\n".join(lines)


def _format_dsv(results: List[SearchResult], delimiter: str) -> str:
    """Render results as delimiter-separated values (CSV or TSV)."""
    # Determine which optional columns are populated
    has_content  = any(r.content          for r in results)
    has_download = any(r.downloaded_file  for r in results)

    fieldnames = ["#", "title", "url", "snippet"]
    if has_content:
        fieldnames += ["page_title", "content", "content_length", "truncated"]
    if has_download:
        fieldnames += ["downloaded_file", "file_size"]

    buf = io.StringIO()
    writer = csv.DictWriter(
        buf,
        fieldnames=fieldnames,
        delimiter=delimiter,
        extrasaction="ignore",
        lineterminator="\n",
    )
    writer.writeheader()

    for i, r in enumerate(results, 1):
        row = {
            "#":              i,
            "title":          r.title,
            "url":            r.url,
            "snippet":        r.snippet,
            "page_title":     r.page_title,
            "content":        r.content,
            "content_length": r.content_length,
            "truncated":      r.truncated,
            "downloaded_file": r.downloaded_file,
            "file_size":       r.file_size,
        }
        writer.writerow(row)

    return buf.getvalue().rstrip("\n")

def _clean_html(raw: str) -> str:
    """Strip HTML tags and normalise whitespace from a fragment."""
    text = re.sub(r"<[^>]+>", " ", raw)
    text = re.sub(r"&nbsp;", " ", text)
    text = re.sub(r"&amp;",  "&", text)
    text = re.sub(r"&lt;",   "<", text)
    text = re.sub(r"&gt;",   ">", text)
    text = re.sub(r"&quot;", '"', text)
    text = re.sub(r"&#39;",  "'", text)
    text = re.sub(r"&[a-z]+;", "", text)   # remaining named entities
    text = re.sub(r"&#\d+;",   "", text)   # numeric entities
    return re.sub(r"\s+", " ", text).strip()

def search(
    # ── Required ──────────────────────────────────────────────────────────────
    query:          str,
    # ── Core options ──────────────────────────────────────────────────────────
    limit:          str = str(DEFAULT_LIMIT),
    backend:        str = DEFAULT_BACKEND,
    format:         str = DEFAULT_FORMAT,          # noqa: A002 (shadows builtin intentionally)
    output:         str = "",
    type:           str = DEFAULT_TYPE,            # noqa: A002
    depth:          str = "1",
    timeout:        str = str(DEFAULT_TIMEOUT),
    delay:          str = str(DEFAULT_DELAY),
    proxy:          str = "",
    user_agent:     str = "",
    header:         Union[str, List[str]] = "",
    content_length: str = str(DEFAULT_CONTENT_LENGTH),
    threads:        str = str(DEFAULT_THREADS),
    safe:           str = DEFAULT_SAFE,
    region:         str = DEFAULT_REGION,
    lang:           str = DEFAULT_LANG,
    searx_url:      str = DEFAULT_SEARX_URL,
    max_crawl_size: str = str(DEFAULT_MAX_CRAWL_SIZE),
    # ── Flags ─────────────────────────────────────────────────────────────────
    crawl:          str = "false",
    download:       str = "false",
    verbose:        str = "false",
    no_robots:      str = "false",
    json_output:    str = "false",
    quiet:          str = "false",
    no_verify:      str = "false",
    dedupe:         str = "false",
    no_anti_bot:    str = "false",
    proxy_rotation: str = "false",
) -> Dict[str, Any]:
    """
    Execute a web search and optionally enrich results with page content.

    Parameters map 1-to-1 with the argc @option/@flag declarations at the
    top of this file so aichat / build-declarations.py can auto-generate
    the tool schema without any custom parsing logic.

    Returns a dict with keys:
      success    (bool)
      results    (List[dict])  — only on success
      count      (int)         — only on success
      formatted  (str)         — only on success
      error      (str)         — only on failure
      duration_ms (float)
    """
    # ---- Convert string arguments to proper Python types ----
    def _to_bool(v: str) -> bool:
        return str(v).lower() in ("true", "1", "yes", "on")

    def _to_int(v: str, default: int = 0) -> int:
        try:
            return int(v)
        except Exception:
            return default

    def _to_float(v: str, default: float = 0.0) -> float:
        try:
            return float(v)
        except Exception:
            return default

    # Core options
    limit = _to_int(limit, DEFAULT_LIMIT)
    depth = _to_int(depth, 1)
    timeout = _to_float(timeout, float(DEFAULT_TIMEOUT))
    delay = _to_float(delay, float(DEFAULT_DELAY))
    content_length = _to_int(content_length, DEFAULT_CONTENT_LENGTH)
    threads = _to_int(threads, DEFAULT_THREADS)
    max_crawl_size = _to_int(max_crawl_size, DEFAULT_MAX_CRAWL_SIZE)

    # Flags (boolean)
    crawl = _to_bool(crawl)
    download = _to_bool(download)
    verbose = _to_bool(verbose)
    no_robots = _to_bool(no_robots)
    json_output = _to_bool(json_output)
    quiet = _to_bool(quiet)
    no_verify = _to_bool(no_verify)
    dedupe = _to_bool(dedupe)
    no_anti_bot = _to_bool(no_anti_bot)
    proxy_rotation = _to_bool(proxy_rotation)

    # Header handling
    if isinstance(header, str):
        header = [h.strip() for h in header.split(",") if h.strip()]
    elif not isinstance(header, list):
        header = []

    t0 = time.perf_counter()

    # ------------------------------------------------------------------
    # Validate inputs
    # ------------------------------------------------------------------
    query = query.strip()
    if not query:
        return _error("query cannot be empty", t0)

    if backend not in VALID_BACKENDS:
        return _error(
            f"Invalid backend '{backend}'. Valid: {', '.join(VALID_BACKENDS)}", t0
        )
    if format not in VALID_FORMATS:
        return _error(
            f"Invalid format '{format}'. Valid: {', '.join(VALID_FORMATS)}", t0
        )
    if type not in VALID_TYPES:
        return _error(
            f"Invalid type '{type}'. Valid: {', '.join(VALID_TYPES)}", t0
        )
    if safe not in VALID_SAFE:
        return _error(
            f"Invalid safe level '{safe}'. Valid: {', '.join(VALID_SAFE)}", t0
        )
    if limit < 1:
        return _error("limit must be >= 1", t0)
    if timeout < 1:
        return _error("timeout must be >= 1", t0)
    if threads < 1:
        return _error("threads must be >= 1", t0)
    if crawl and download:
        return _error("--crawl and --download are mutually exclusive", t0)

    # ------------------------------------------------------------------
    # Build config
    # ------------------------------------------------------------------
    cfg = SearchConfig(
        query          = query,
        limit          = limit,
        backend        = backend,
        fmt            = format,
        output_file    = output,
        search_type    = type,
        depth          = depth,
        timeout        = timeout,
        delay          = delay,
        proxy          = proxy,
        user_agent     = user_agent,
        headers        = list(header) if header else [],
        content_length = content_length,
        threads        = threads,
        safe           = safe,
        region         = region,
        lang           = lang,
        searx_url      = searx_url,
        max_crawl_size = max_crawl_size,
        do_crawl       = crawl,
        do_download    = download,
        verbose        = verbose,
        no_robots      = no_robots,
        force_json     = json_output,
        quiet          = quiet,
        no_verify      = no_verify,
        dedupe         = dedupe,
        # Enhanced anti-bot features (can be disabled with --no-anti-bot)
        use_realistic_headers = not no_anti_bot,
        enable_session_persistence = not no_anti_bot,
        circuit_breaker_threshold = 5 if not no_anti_bot else 1,
        max_retry_attempts = 3 if not no_anti_bot else 1,
        proxy_rotation_enabled = proxy_rotation,
        # Additional anti-bot settings
        anti_bot_delay_range = (0.5, 2.0) if not no_anti_bot else (0.1, 0.5),
        enable_request_fingerprinting = not no_anti_bot,
        enable_adaptive_timing = not no_anti_bot,
        enable_backend_optimization = not no_anti_bot,
    )

    if verbose:
        logging.basicConfig(
            level  = logging.DEBUG,
            format = "%(levelname)s %(name)s: %(message)s",
            stream = sys.stderr,
        )

    if not quiet:
        _cprint(
            f"[→] Searching '{query}' via {backend.upper()} "
            f"(limit={limit}, format={format})",
            Fore.CYAN,
        )

    # ------------------------------------------------------------------
    # Search
    # ------------------------------------------------------------------
    try:
        results = search(cfg)
    except Exception as exc:
        return _error(f"Search failed: {exc}", t0)

    if not results:
        # Provide more helpful error message with suggestions
        return _error(
            f"No results found for: {query}. "
            f"Try: different keywords, --backend bing, --backend searx, or --verbose for debugging", 
            t0
        )

    if not quiet:
        _cprint(f"[✓] Found {len(results)} results", Fore.GREEN)

    # ------------------------------------------------------------------
    # Enrich (crawl / download)
    # ------------------------------------------------------------------
    if crawl or download:
        action = "Crawling" if crawl else "Downloading"
        if not quiet:
            _cprint(f"[→] {action} {len(results)} pages...", Fore.CYAN)
        try:
            results = _enrich_results(results, cfg)
        except Exception as exc:
            logger.warning("Enrichment error: %s", exc)

    # ------------------------------------------------------------------
    # Format
    # ------------------------------------------------------------------
    try:
        formatted = format_results(results, format)
    except Exception as exc:
        return _error(f"Formatting failed: {exc}", t0)

    # ------------------------------------------------------------------
    # Output
    # ------------------------------------------------------------------
    if output:
        try:
            Path(output).write_text(formatted, encoding="utf-8")
            if not quiet:
                _cprint(f"[✓] Results saved to: {output}", Fore.GREEN)
        except OSError as exc:
            return _error(f"Failed to write output file: {exc}", t0)
    else:
        # Check LLM_OUTPUT env var (aichat pipe integration)
        llm_out = os.environ.get("LLM_OUTPUT", "")
        if llm_out and llm_out != "/dev/fd/1":
            try:
                Path(llm_out).write_text(formatted, encoding="utf-8")
            except OSError as exc:
                logger.debug("LLM_OUTPUT write failed: %s", exc)
        else:
            print(formatted)

    ms = round((time.perf_counter() - t0) * 1000, 2)
    return {
        "success":     True,
        "results":     [r.to_dict() for r in results],
        "count":       len(results),
        "formatted":   formatted,
        "duration_ms": ms,
    }


def _error(msg: str, t0: float) -> Dict[str, Any]:
    """Build a standardised error response dict."""
    ms = round((time.perf_counter() - t0) * 1000, 2)
    logger.error(msg)
    _cprint(f"[✗] {msg}", Fore.RED)
    return {"success": False, "error": msg, "duration_ms": ms}
